pipeline {
    agent any
    
    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/Onkar27vk/terraform_pipeline.git'
            }
        }
        
        stage('Terraform Init') {
            steps {
                sh 'terraform init'
             }
        }
        
        stage('Terraform Plan') {
            steps {
                sh 'terraform plan'
            }
        }
        
        stage('Terraform Apply') {
            steps {
                sh 'terraform apply -auto-approve'
                }
            }
        }
    }
